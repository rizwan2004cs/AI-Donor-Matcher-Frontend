# Feature 7.1 — Navbar

> **Branch:** `feat/navbar`
> **File:** `src/components/Navbar.jsx`
> **Depends on:** AuthContext

---

## Overview

A role-based navigation bar. Shows different links for Guest, Donor, NGO, and Admin users. Includes the app logo and a logout button for authenticated users.

## Component Specification

### Props

None — reads from AuthContext.

### Full Component

```jsx
import { Link, useNavigate } from "react-router-dom";
import { useAuth } from "../auth/AuthContext";

export default function Navbar() {
  const { user, logout } = useAuth();
  const navigate = useNavigate();

  const handleLogout = () => {
    logout();
    navigate("/login");
  };

  return (
    <nav className="bg-white shadow px-6 py-3 flex items-center justify-between">
      {/* Logo */}
      <Link to="/" className="text-xl font-bold text-indigo-600">
        AI Donation Matcher
      </Link>

      {/* Navigation links */}
      <div className="flex items-center gap-4">
        {/* Guest */}
        {!user && (
          <>
            <Link to="/login" className="text-sm text-gray-600 hover:text-indigo-600">
              Login
            </Link>
            <Link to="/register" className="text-sm bg-indigo-600 text-white px-4 py-1.5 rounded hover:bg-indigo-700">
              Register
            </Link>
          </>
        )}

        {/* Donor */}
        {user?.role === "DONOR" && (
          <>
            <Link to="/" className="text-sm text-gray-600 hover:text-indigo-600">
              Discover
            </Link>
            <Link to="/dashboard/donor" className="text-sm text-gray-600 hover:text-indigo-600">
              My Pledges
            </Link>
          </>
        )}

        {/* NGO */}
        {user?.role === "NGO" && (
          <Link to="/dashboard/ngo" className="text-sm text-gray-600 hover:text-indigo-600">
            Dashboard
          </Link>
        )}

        {/* Admin */}
        {user?.role === "ADMIN" && (
          <Link to="/dashboard/admin" className="text-sm text-gray-600 hover:text-indigo-600">
            Admin Panel
          </Link>
        )}

        {/* Authenticated — user info + logout */}
        {user && (
          <>
            <span className="text-sm text-gray-500">{user.fullName}</span>
            <button
              onClick={handleLogout}
              className="text-sm bg-red-500 text-white px-4 py-1.5 rounded hover:bg-red-600"
            >
              Logout
            </button>
          </>
        )}
      </div>
    </nav>
  );
}
```

---

## Role Variants

| Role | Links Shown |
|------|-------------|
| Guest (no user) | Login, Register |
| Donor | Discover, My Pledges, [name], Logout |
| NGO | Dashboard, [name], Logout |
| Admin | Admin Panel, [name], Logout |
