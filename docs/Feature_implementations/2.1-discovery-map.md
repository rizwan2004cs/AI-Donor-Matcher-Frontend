## Overview

The main donor-facing screen. A full-width Leaflet map with colour-coded pins for each verified NGO. Includes a filter bar at the top and a side list on the right. Requests browser geolocation on mount.

## Component Specification

### State

```js
const [ngos, setNgos] = useState([]);
const [loading, setLoading] = useState(true);
const [error, setError] = useState(null);
const [userLat, setUserLat] = useState(null);
const [userLng, setUserLng] = useState(null);
const [filters, setFilters] = useState({
  search: "",
  category: "",
  radius: 10,
});
const [expanded, setExpanded] = useState(false); // nationwide fallback active
```

### Geolocation on Mount

```js
useEffect(() => {
  navigator.geolocation.getCurrentPosition(
    (pos) => {
      setUserLat(pos.coords.latitude);
      setUserLng(pos.coords.longitude);
    },
    () => {
      // Denied or unavailable — default to Chennai
      setUserLat(13.0827);
      setUserLng(80.2707);
    }
  );
}, []);
```

### Fetch NGOs (triggered when location or filters change)

```js
useEffect(() => {
  if (userLat === null) return;

  const fetchNgos = async () => {
    setLoading(true);
    setExpanded(false);
    try {
      const params = { lat: userLat, lng: userLng, radius: filters.radius };
      if (filters.category) params.category = filters.category;
      if (filters.search) params.search = filters.search;

      const res = await api.get("/api/ngos", { params });

      if (res.data.length === 0) {
        // Auto-expand nationwide (see 2.4)
        const { radius, ...nationwideParams } = params;
        const fallback = await api.get("/api/ngos", { params: nationwideParams });
        setNgos(fallback.data);
        setExpanded(true);
      } else {
        setNgos(res.data);
      }
    } catch (err) {
      setError(err.response?.data?.message || "Failed to load NGOs");
    } finally {
      setLoading(false);
    }
  };

  fetchNgos();
}, [userLat, userLng, filters]);
```

### Layout JSX

```jsx
import { MapContainer, TileLayer, Marker, Popup } from "react-leaflet";
import Navbar from "../components/Navbar";
import { createCategoryIcon } from "../components/CategoryPin";

return (
  <div className="h-screen flex flex-col">
    <Navbar />
    <FilterBar filters={filters} setFilters={setFilters} />

    {expanded && (
      <div className="bg-yellow-100 text-yellow-800 text-sm px-4 py-2 text-center">
        No NGOs found nearby. Showing all available NGOs.
      </div>
    )}

    <div className="flex flex-1 overflow-hidden">
      {/* Map — left panel */}
      <div className="flex-1">
        <MapContainer
          center={[userLat || 13.0827, userLng || 80.2707]}
          zoom={12}
          className="h-full w-full"
        >
          <TileLayer url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png" />
          {ngos.map((ngo) => (
            <Marker
              key={ngo.id}
              position={[ngo.lat, ngo.lng]}
              icon={createCategoryIcon(ngo.pinCategory)}
            >
              {/* Popup — see 2.5 */}
              <Popup>...</Popup>
            </Marker>
          ))}
        </MapContainer>
      </div>

      {/* Side list — right panel (see 2.3) */}
      <SideList ngos={ngos} />
    </div>
  </div>
);
```
