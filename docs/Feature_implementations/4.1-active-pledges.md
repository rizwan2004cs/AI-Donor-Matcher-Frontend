### Fetch

```js
useEffect(() => {
  api.get("/api/pledges/my/active")
    .then((res) => setPledges(res.data))
    .catch((err) => setError(err.response?.data?.message || "Failed to load pledges"))
    .finally(() => setLoading(false));
}, []);
```

### Cancel Handler (Optimistic)

```js
const handleCancel = async (pledgeId) => {
  if (!confirm("Cancel this pledge?")) return;

  // Optimistic removal
  setPledges((prev) => prev.filter((p) => p.pledgeId !== pledgeId));

  try {
    await api.delete(`/api/pledges/${pledgeId}`);
  } catch {
    // Re-fetch on failure
    const res = await api.get("/api/pledges/my/active");
    setPledges(res.data);
    alert("Failed to cancel pledge");
  }
};
```

### Expiry Countdown Helper

```js
function useCountdown(expiresAt) {
  const [timeLeft, setTimeLeft] = useState("");

  useEffect(() => {
    const tick = () => {
      const secs = Math.max(0, Math.floor((new Date(expiresAt) - Date.now()) / 1000));
      const h = Math.floor(secs / 3600);
      const m = Math.floor((secs % 3600) / 60);
      setTimeLeft(secs <= 0 ? "Expired" : `${h}h ${m}m`);
    };
    tick();
    const id = setInterval(tick, 60000); // update every minute
    return () => clearInterval(id);
  }, [expiresAt]);

  return timeLeft;
}
```

### Pledge Card Component

```jsx
function PledgeCard({ pledge, onCancel, onNavigate }) {
  const timeLeft = useCountdown(pledge.expiresAt);

  return (
    <div className="border rounded-lg p-4 flex items-center justify-between">
      <div className="flex items-center gap-3">
        <span
          className="w-3 h-3 rounded-full"
          style={{ backgroundColor: CATEGORY_COLORS[pledge.category] }}
        />
        <div>
          <p className="font-medium">{pledge.itemName} × {pledge.quantity}</p>
          <p className="text-sm text-gray-500">{pledge.ngoName} · {pledge.distanceKm} km</p>
        </div>
      </div>

      <div className="flex items-center gap-3">
        <span className={`text-sm ${timeLeft === "Expired" ? "text-red-600" : "text-orange-500"}`}>
          ⏱ {timeLeft}
        </span>
        <button
          onClick={() => onNavigate(pledge.pledgeId)}
          className="bg-blue-600 text-white px-3 py-1.5 rounded text-sm hover:bg-blue-700"
        >
          Navigate
        </button>
        <button
          onClick={() => onCancel(pledge.pledgeId)}
          className="bg-red-100 text-red-600 px-3 py-1.5 rounded text-sm hover:bg-red-200"
        >
          Cancel
        </button>
      </div>
    </div>
  );
}
```
