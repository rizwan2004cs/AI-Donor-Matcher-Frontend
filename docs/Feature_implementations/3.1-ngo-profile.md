Public-facing NGO profile page. Displays NGO photo, name, trust score, address, contact, active needs with progress bars, fulfilled history, and a report button.

### Report NGO

```
POST /api/ngos/{id}/report
Content-Type: application/json

Request Body:
{ "reason": "FRAUD" | "INACTIVE" | "MISLEADING" | "OTHER" }

Response (200):
{ "message": "Report submitted successfully" }
```

## Component Specification

### State

```js
const { id } = useParams();
const { user } = useAuth();
const navigate = useNavigate();

const [ngo, setNgo] = useState(null);
const [loading, setLoading] = useState(true);
const [error, setError] = useState(null);
const [showReportModal, setShowReportModal] = useState(false);
```

### Fetch on Mount

```js
useEffect(() => {
  api.get(`/api/ngos/${id}`)
    .then((res) => setNgo(res.data))
    .catch((err) => setError(err.response?.data?.message || "Failed to load NGO"))
    .finally(() => setLoading(false));
}, [id]);
```

### Pledge Button Logic

```jsx
const handlePledge = (needId) => {
  if (!user) {
    navigate("/login");
    return;
  }
  if (!user.emailVerified) {
    // Show tooltip: "Verify your email to pledge."
    return;
  }
  navigate(`/pledge/${needId}`);
};
```

### Layout JSX

```jsx
return (
  <div>
    <Navbar />
    <div className="max-w-3xl mx-auto p-6">
      {/* Header */}
      <div className="flex items-center gap-4 mb-6">
        {ngo.photoUrl ? (
          <img src={ngo.photoUrl} className="w-20 h-20 rounded-full object-cover" />
        ) : (
          <div className="w-20 h-20 rounded-full bg-gray-200 flex items-center justify-center text-2xl font-bold text-gray-500">
            {ngo.name.charAt(0)}
          </div>
        )}
        <div>
          <h1 className="text-2xl font-bold">{ngo.name}</h1>
          <TrustBadge score={ngo.trustScore} label={ngo.trustLabel} />
          <p className="text-sm text-gray-500 mt-1">Verified since {ngo.verifiedSince}</p>
        </div>
      </div>

      {/* Details */}
      <p className="text-gray-700 mb-4">{ngo.description}</p>
      <p className="text-sm text-gray-500">📍 {ngo.address}</p>
      <p className="text-sm text-gray-500">✉️ {ngo.contactEmail}</p>

      {/* Active Needs */}
      <h2 className="text-lg font-semibold mt-8 mb-3">Active Needs</h2>
      {ngo.activeNeeds.map((need) => {
        const remaining = need.quantityRequired - need.quantityPledged;
        return (
          <div key={need.id} className="border rounded p-4 mb-3">
            <div className="flex items-center justify-between">
              <div className="flex items-center gap-2">
                <span
                  className="w-3 h-3 rounded-full"
                  style={{ backgroundColor: CATEGORY_COLORS[need.category] }}
                />
                <span className="font-medium">{need.itemName}</span>
                {need.urgency === "URGENT" && (
                  <span className="text-xs bg-red-100 text-red-600 px-2 py-0.5 rounded">
                    Urgent
                  </span>
                )}
              </div>
              <span className="text-sm text-gray-500">{remaining} remaining</span>
            </div>

            <NeedProgressBar
              pledged={need.quantityPledged}
              required={need.quantityRequired}
            />

            <button
              onClick={() => handlePledge(need.id)}
              disabled={!user?.emailVerified && !!user}
              className="mt-2 bg-blue-600 text-white px-4 py-1.5 rounded text-sm hover:bg-blue-700 disabled:opacity-50"
              title={user && !user.emailVerified ? "Verify your email to pledge." : ""}
            >
              Pledge This Item
            </button>
          </div>
        );
      })}

      {/* Fulfilled History */}
      <h2 className="text-lg font-semibold mt-8 mb-3">Past Fulfilled Needs</h2>
      {ngo.fulfilledHistory.map((item, i) => (
        <div key={i} className="flex items-center gap-2 py-2 border-b text-sm">
          <span className="text-green-500">✅</span>
          <span>{item.itemName}</span>
          <span className="text-gray-400">× {item.quantity}</span>
          <span className="text-gray-400 ml-auto">{item.fulfilledDate}</span>
        </div>
      ))}

      {/* Report */}
      <button
        onClick={() => setShowReportModal(true)}
        className="mt-8 text-sm text-red-500 hover:underline"
      >
        ⚑ Report this organisation
      </button>
    </div>
  </div>
);
```
