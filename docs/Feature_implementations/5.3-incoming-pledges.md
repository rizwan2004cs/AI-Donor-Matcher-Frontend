## Component Specification

### State

```js
const [pledges, setPledges] = useState([]);
const [pledgesLoading, setPledgesLoading] = useState(true);
```

### Fetch

```js
useEffect(() => {
  api.get("/api/ngo/my/pledges")
    .then((res) => setPledges(res.data))
    .catch(() => {})
    .finally(() => setPledgesLoading(false));
}, []);
```

### Time Ago Helper

```js
function timeAgo(dateString) {
  const seconds = Math.floor((Date.now() - new Date(dateString)) / 1000);
  if (seconds < 60) return "just now";
  const mins = Math.floor(seconds / 60);
  if (mins < 60) return `${mins}m ago`;
  const hours = Math.floor(mins / 60);
  if (hours < 24) return `${hours}h ago`;
  const days = Math.floor(hours / 24);
  return `${days}d ago`;
}
```

### Layout JSX

```jsx
<div className="mt-8">
  <h2 className="text-lg font-semibold mb-3">Incoming Pledges</h2>

  {pledges.length === 0 && (
    <p className="text-gray-400 text-sm">No incoming pledges yet.</p>
  )}

  <div className="space-y-2">
    {pledges.map((p, i) => (
      <div key={i} className="flex items-center justify-between border rounded p-3">
        <div>
          <p className="font-medium text-sm">{p.donorName}</p>
          <p className="text-xs text-gray-500">{p.donorEmail}</p>
        </div>
        <div className="text-right">
          <p className="text-sm">
            {p.itemName} × <strong>{p.quantity}</strong>
          </p>
          <p className="text-xs text-gray-400">{timeAgo(p.pledgedAt)}</p>
        </div>
      </div>
    ))}
  </div>
</div>
```
