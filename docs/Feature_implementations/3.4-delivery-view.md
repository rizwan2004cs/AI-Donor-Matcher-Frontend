## Overview

Post-pledge navigation screen. Displays an OSRM-routed polyline on a Leaflet map between the donor's current location and the NGO drop-off point. Shows ETA, distance, expiry countdown, NGO contact, and a cancel button.

## API Contract

### Fetch Pledge Data (if not passed via router state)

```
GET /api/pledges/{pledgeId}

Response (200):
{
  "pledgeId": 42,
  "ngoLat": 13.0450,
  "ngoLng": 80.2350,
  "ngoAddress": "12 Gandhi Road, Chennai",
  "ngoEmail": "hope@ngo.org",
  "itemName": "Rice (10kg bags)",
  "quantity": 5,
  "expiresAt": "2026-03-07T14:30:00Z",
  "status": "ACTIVE"
}
```

### Cancel Pledge

```
DELETE /api/pledges/{pledgeId}

Response (200):
{ "message": "Pledge cancelled" }
```

### OSRM Route

```
GET https://router.project-osrm.org/route/v1/driving/{donorLng},{donorLat};{ngoLng},{ngoLat}?overview=full&geometries=geojson

Response: (OSRM standard format)
{
  "routes": [{
    "distance": 5430.2,
    "duration": 812.5,
    "geometry": {
      "type": "LineString",
      "coordinates": [[80.27, 13.08], [80.25, 13.06], ...]
    }
  }]
}
```

## Component Specification

### State

```js
const { pledgeId } = useParams();
const location = useLocation();
const navigate = useNavigate();

const [pledge, setPledge] = useState(location.state || null);
const [donorPos, setDonorPos] = useState(null);
const [routeCoords, setRouteCoords] = useState([]);
const [distanceKm, setDistanceKm] = useState(null);
const [durationMins, setDurationMins] = useState(null);
const [timeLeft, setTimeLeft] = useState("");
const [loading, setLoading] = useState(true);
const [cancelling, setCancelling] = useState(false);
```

### Fetch Pledge (if not in router state)

```js
useEffect(() => {
  if (pledge) {
    setLoading(false);
    return;
  }
  api.get(`/api/pledges/${pledgeId}`)
    .then((res) => setPledge(res.data))
    .catch(() => navigate("/dashboard/donor"))
    .finally(() => setLoading(false));
}, [pledgeId]);
```

### Geolocation + OSRM Route

```js
const OSRM = import.meta.env.VITE_OSRM_URL;

useEffect(() => {
  if (!pledge) return;

  navigator.geolocation.getCurrentPosition(
    async (pos) => {
      const donorLat = pos.coords.latitude;
      const donorLng = pos.coords.longitude;
      setDonorPos([donorLat, donorLng]);

      try {
        const url = `${OSRM}/route/v1/driving/${donorLng},${donorLat};${pledge.ngoLng},${pledge.ngoLat}?overview=full&geometries=geojson`;
        const res = await fetch(url);
        const data = await res.json();
        const route = data.routes[0];

        setRouteCoords(
          route.geometry.coordinates.map(([lng, lat]) => [lat, lng])
        );
        setDistanceKm((route.distance / 1000).toFixed(1));
        setDurationMins(Math.ceil(route.duration / 60));
      } catch {
        // Fallback: draw straight line
        setRouteCoords([
          [donorLat, donorLng],
          [pledge.ngoLat, pledge.ngoLng],
        ]);
      }
    },
    () => {
      // Geolocation denied — centre on NGO
      setDonorPos(null);
    }
  );
}, [pledge]);
```

### Expiry Countdown

```js
useEffect(() => {
  if (!pledge?.expiresAt) return;

  const interval = setInterval(() => {
    const secondsLeft = Math.max(
      0,
      Math.floor((new Date(pledge.expiresAt) - Date.now()) / 1000)
    );
    const h = Math.floor(secondsLeft / 3600);
    const m = Math.floor((secondsLeft % 3600) / 60);
    setTimeLeft(`${h}h ${m}m`);

    if (secondsLeft <= 0) {
      clearInterval(interval);
      setTimeLeft("Expired");
    }
  }, 1000);

  return () => clearInterval(interval);
}, [pledge]);
```

### Cancel Handler

```js
const handleCancel = async () => {
  if (!confirm("Cancel this pledge?")) return;
  setCancelling(true);
  try {
    await api.delete(`/api/pledges/${pledgeId}`);
    navigate("/dashboard/donor");
  } catch (err) {
    alert(err.response?.data?.message || "Failed to cancel");
  } finally {
    setCancelling(false);
  }
};
```

### Layout JSX

```jsx
import { MapContainer, TileLayer, Marker, Polyline, Popup } from "react-leaflet";

return (
  <div className="h-screen flex flex-col">
    <Navbar />

    {/* ETA bar */}
    <div className="bg-blue-50 px-4 py-3 flex items-center justify-between text-sm">
      <span>📍 {distanceKm} km · {durationMins} min</span>
      <span className={timeLeft === "Expired" ? "text-red-600 font-bold" : "text-orange-600"}>
        ⏱ {timeLeft}
      </span>
    </div>

    {/* Map */}
    <div className="flex-1">
      <MapContainer
        center={donorPos || [pledge.ngoLat, pledge.ngoLng]}
        zoom={13}
        className="h-full w-full"
      >
        <TileLayer url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png" />

        {donorPos && (
          <Marker position={donorPos}>
            <Popup>📍 You are here</Popup>
          </Marker>
        )}

        <Marker position={[pledge.ngoLat, pledge.ngoLng]}>
          <Popup>🏢 Drop-off: {pledge.ngoAddress}</Popup>
        </Marker>

        {routeCoords.length > 0 && (
          <Polyline positions={routeCoords} color="blue" weight={4} />
        )}
      </MapContainer>
    </div>

    {/* Info panel */}
    <div className="bg-white border-t p-4">
      <div className="flex items-center justify-between mb-2">
        <div>
          <p className="font-semibold">{pledge.itemName} × {pledge.quantity}</p>
          <p className="text-sm text-gray-500">📍 {pledge.ngoAddress}</p>
          <p className="text-sm text-gray-500">✉️ {pledge.ngoEmail}</p>
        </div>
      </div>
      <button
        onClick={handleCancel}
        disabled={cancelling}
        className="w-full bg-red-500 text-white py-2 rounded hover:bg-red-600 disabled:opacity-50"
      >
        {cancelling ? "Cancelling..." : "Cancel Pledge"}
      </button>
    </div>
  </div>
);
```
