```
Content-Type: application/json

Request Body:
{
  "email": "rahul@gmail.com",
  "password": "SecurePass123"
}

Response (200):
{
  "token": "eyJhbGciOiJIUzI1NiJ9...",

    "id": 1,
    "fullName": "Rahul Kumar",
    "email": "rahul@gmail.com",
    "role": "DONOR",
    "emailVerified": true
  }
}

Error (401):
{ "message": "Invalid email or password" }

Error (403):
{ "message": "Account suspended" }
```

## Component Specification

### State

```js
const [email, setEmail] = useState("");
const [password, setPassword] = useState("");
const [error, setError] = useState(null);
const [loading, setLoading] = useState(false);
```

### Submit Handler

```js
import { useAuth } from "../auth/AuthContext";
import { useNavigate } from "react-router-dom";

const { login } = useAuth();
const navigate = useNavigate();

const handleSubmit = async (e) => {
  e.preventDefault();
  setError(null);
  setLoading(true);

  try {
    const res = await api.post("/api/auth/login", { email, password });
    const { token, user } = res.data;
    login(user, token);

    // Role-based redirect
    switch (user.role) {
      case "DONOR":
        navigate("/map");
        break;
      case "NGO":
        navigate(user.profileComplete ? "/ngo/dashboard" : "/ngo/complete-profile");
        break;
      case "ADMIN":
        navigate("/admin/dashboard");
        break;
      default:
        navigate("/");
    }
  } catch (err) {
    setError(err.response?.data?.message || "Login failed");
  } finally {
    setLoading(false);
  }
};
```

### NGO Profile Completion Check

When an NGO logs in, check `user.profileComplete`:
- If `true` → redirect to `/ngo/dashboard`
- If `false` → redirect to `/ngo/complete-profile`

This ensures NGOs complete their profile before accessing the dashboard.

## What `login()` Does

From `src/auth/AuthContext.jsx`:

```js
const login = (userData, token) => {
  localStorage.setItem("token", token);
  localStorage.setItem("user", JSON.stringify(userData));
  setUser(userData);
};
```

This triggers a re-render throughout the app. The Navbar updates to show role-specific links. ProtectedRoute components now allow access.
