## Overview

A Leaflet `DivIcon` factory that creates colour-coded map pins based on donation category. Each category has a distinct colour so donors can scan the map at a glance.

## Component Specification

### categoryColors.js

```js
const categoryColors = {
  FOOD:      "#EF4444",
  CLOTHING:  "#3B82F6",
  MEDICINE:  "#22C55E",
  EDUCATION: "#EAB308",
  HOUSEHOLD: "#F97316",
  OTHER:     "#A855F7",
};

export default categoryColors;
```

### createCategoryIcon() Factory

```jsx
import L from "leaflet";
import categoryColors from "../utils/categoryColors";

export function createCategoryIcon(category) {
  const color = categoryColors[category] || categoryColors.OTHER;

  return L.divIcon({
    className: "",                          // remove Leaflet default box-shadow
    iconSize: [28, 28],
    iconAnchor: [14, 28],                   // bottom-center of the circle
    popupAnchor: [0, -30],
    html: `
      <div style="
        width:28px; height:28px; border-radius:50%;
        background:${color}; border:2px solid white;
        box-shadow:0 2px 6px rgba(0,0,0,0.3);
      "></div>
    `,
  });
}
```

## Usage in DiscoveryMap

```jsx
import { Marker, Popup } from "react-leaflet";
import { createCategoryIcon } from "../components/CategoryPin";

{needs.map((n) => (
  <Marker
    key={n.id}
    position={[n.lat, n.lng]}
    icon={createCategoryIcon(n.category)}
  >
    <Popup>
      <strong>{n.ngoName}</strong><br />
      {n.itemName} — {n.category}
    </Popup>
  </Marker>
))}
```

## Behaviour Notes

- `L.divIcon` avoids an external image; works offline after first load.
- `className: ""` removes the default Leaflet border/background.
- `iconAnchor [14, 28]` ensures the pin points at the exact lat/lng.
- Unknown categories fall back to `OTHER` (purple).
