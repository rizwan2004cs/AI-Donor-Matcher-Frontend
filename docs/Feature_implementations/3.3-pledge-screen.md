## Overview

A form allowing verified donors to pledge a quantity of items toward an NGO's active need. Quantity is clamped between 1 and the remaining amount. On success, redirects to the delivery view.

## API Contract

```
GET /api/needs/{needId}

Response (200):
{

```
POST /api/pledges

## Component Specification

### State

```js
const { needId } = useParams();
const navigate = useNavigate();

const [need, setNeed] = useState(null);
const [qty, setQty] = useState(1);
const [loading, setLoading] = useState(true);
const [submitting, setSubmitting] = useState(false);
const [error, setError] = useState(null);
```

### Fetch on Mount

```js
useEffect(() => {
  api.get(`/api/needs/${needId}`)
    .then((res) => setNeed(res.data))
    .catch((err) => setError(err.response?.data?.message || "Failed to load need"))
    .finally(() => setLoading(false));
}, [needId]);
```

### Quantity Selector Logic

```js
const remaining = need.quantityRequired - need.quantityPledged;

const increment = () => setQty((q) => Math.min(q + 1, remaining));
const decrement = () => setQty((q) => Math.max(q - 1, 1));
```

### Submit Handler

```js
const handleConfirm = async () => {
  setSubmitting(true);
  setError(null);

  try {
    const res = await api.post("/api/pledges", { needId: need.id, quantity: qty });
    // Navigate to delivery view with pledge data
    navigate(`/delivery/${res.data.pledgeId}`, { state: res.data });
  } catch (err) {
    setError(err.response?.data?.message || "Failed to create pledge");
  } finally {
    setSubmitting(false);
  }
};
```

### Layout JSX

```jsx
return (
  <div>
    <Navbar />
    <div className="max-w-md mx-auto p-6">
      <h1 className="text-xl font-bold mb-2">Pledge Donation</h1>

      {/* Need details */}
      <div className="bg-gray-50 rounded p-4 mb-6">
        <p className="text-sm text-gray-500">NGO: <strong>{need.ngoName}</strong></p>
        <p className="text-lg font-semibold mt-1">{need.itemName}</p>
        <div className="flex items-center gap-2 mt-1">
          <span
            className="w-3 h-3 rounded-full"
            style={{ backgroundColor: CATEGORY_COLORS[need.category] }}
          />
          <span className="text-sm text-gray-600">{need.category}</span>
          {need.urgency === "URGENT" && (
            <span className="text-xs bg-red-100 text-red-600 px-2 py-0.5 rounded">
              Urgent
            </span>
          )}
        </div>
        <p className="text-sm text-gray-600 mt-2">
          Total required: {need.quantityRequired} · Pledged: {need.quantityPledged} ·
          <strong> Remaining: {remaining}</strong>
        </p>
      </div>

      {/* Quantity selector */}
      <div className="flex items-center justify-center gap-4 mb-6">
        <button
          onClick={decrement}
          disabled={qty <= 1}
          className="w-10 h-10 rounded-full border text-lg font-bold disabled:opacity-30"
        >
          −
        </button>
        <input
          type="number"
          value={qty}
          onChange={(e) => {
            const v = parseInt(e.target.value) || 1;
            setQty(Math.max(1, Math.min(v, remaining)));
          }}
          className="w-20 text-center text-xl font-bold border rounded py-2"
          min={1}
          max={remaining}
        />
        <button
          onClick={increment}
          disabled={qty >= remaining}
          className="w-10 h-10 rounded-full border text-lg font-bold disabled:opacity-30"
        >
          +
        </button>
      </div>

      {error && <p className="text-red-500 text-sm mb-4">{error}</p>}

      {/* Confirm */}
      <button
        onClick={handleConfirm}
        disabled={submitting}
        className="w-full bg-green-600 text-white py-3 rounded-lg font-medium hover:bg-green-700 disabled:opacity-50"
      >
        {submitting ? "Confirming..." : `Confirm Pledge (${qty} items)`}
      </button>

      <p className="text-xs text-gray-400 text-center mt-3">
        A confirmation email will be sent to you with the NGO address and contact.
      </p>
    </div>
  </div>
);
```
